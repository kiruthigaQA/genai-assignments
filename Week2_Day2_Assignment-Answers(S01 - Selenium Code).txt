import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.chrome.ChromeDriver;

public class LoginTest {

    public static void main(String[] args) {

        WebDriver driver = new ChromeDriver();
        driver.get("http://example.com"); // Replace with actual URL

        // Locate Username field using name
        WebElement username = driver.findElement(By.name("USERNAME"));
        username.sendKeys("your_username");

        // Locate Password field using name
        WebElement password = driver.findElement(By.name("PASSWORD"));
        password.sendKeys("your_password");

        // Locate Login button using class and type
        WebElement loginButton = driver.findElement(By.xpath("//input[@type='submit' and @class='decorativeSubmit']"));
        loginButton.click();

        driver.quit();
    }
}
